## ğŸ’³ 5. Payment Page (Connected to API)

import React, { useState } from 'react';
import { View, Text, TextInput, Button, Alert } from 'react-native';
import axios from 'axios';
import { API_BASE_URL } from '../config';

export default function PaymentScreen() {
  const [studentId, setStudentId] = useState('');
  const [amount, setAmount] = useState('');

  async function handlePayment() {
    try {
      const res = await axios.post(`${API_BASE_URL}/create-invoice`, {
        studentId,
        amount: parseFloat(amount),
      });
      if (res.data.success) {
        Alert.alert('Payment Initiated', 'Invoice created successfully.');
      } else {
        Alert.alert('Error', res.data.message);
      }
    } catch (err) {
      Alert.alert('Error', 'Failed to connect to server.');
    }
  }

  return (
    <View className="p-4">
      <Text className="text-xl font-bold mb-2">Make Payment</Text>
      <TextInput
        placeholder="Student ID"
        className="border p-2 mb-2 rounded"
        value={studentId}
        onChangeText={setStudentId}
      />
      <TextInput
        placeholder="Amount"
        keyboardType="numeric"
        className="border p-2 mb-4 rounded"
        value={amount}
        onChangeText={setAmount}
      />
      <Button title="Pay Now" onPress={handlePayment} />
    </View>
  );
}
